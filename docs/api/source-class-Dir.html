110:     /**
111:      * Find all files within the specified path, that match the pattern (if supplied)
112:      * 
113:      * @param string $path the path to scan
114:      * @param string $regexPattern the regex pattern to filter the find results
115:      * @param boolean $recursive recursively scan through directories
116:      * @return array|boolean false on invalid path
117:      */
118:     private function _findFiles($path, $regexPattern = null, $recursive = false)
119:     {
120:         
121:         if ($this->_exists($path)) {
122:             $files = array();
123:             foreach (new DirectoryIterator($path) as $iteminfo) {
124:                 $pathname = $iteminfo->getPathname();
125:                 if ($iteminfo->isFile() && ($regexPattern === null || (preg_match($regexPattern, $pathname)))) {
126:                     $files[] = $pathname;
127:                 } else if ($iteminfo->isDir() && !$iteminfo->isDot() && $recursive) {
128:                     $dirFiles = $this->_findFiles($pathname, $regexPattern, $recursive);
129:                     if (is_array($dirFiles)) {
130:                         $files = array_merge($files, $dirFiles);
131:                     }
132:                 }
133:             }
134:             return $files;
135:         }
136:         return false;
137:     }
